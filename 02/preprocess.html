def read_json(path: str) -> Dict:
    """jsonファイルを辞書として読み込む"""
    with open(path, "r") as f:
        dct = json.load(f)

    return dct

def save_json(path: str, dct: Dict) -> None:
    """辞書をjson形式で保存する関数．"""
    with open(path, "w") as f:
        json.dump(dct, f, ensure_ascii=False, indent=4)

def read_json_w_filters(path: str, items: List, filters: Dict) -> None:
    """itemsのうち，filtersの条件を満たすもののみを抽出"""
    out = []
    with open(path, "r") as f:
        parse = ijson.items(f, items)
        for item in parse:
            # filtersの条件をすべて満足するもの以外はbreak
            for k, v in filters.items():
                if k not in item.keys():
                    break
                if item[k] not in v:
                    break
            else:
                # breakしなかった場合はoutに追加
                out.append(item)
    return out

def try_mkdirs(path: str) -> None:
    """mkdirsにtry"""
    try:
        os.makedirs(path)
    except FileExistsError as e:
        pass

def format_magazine_name(name: Any) -> str:
    """nameからpublished_nameを取得"""
    for x in name:
        if type(x) is str:
            return x
    raise Exception(f"No magazine name in {name}!")

def format_cols(df: pd.DataFrame, cols_rename: List) -> pd.DataFrame:
    """cols_renameのcolのみを抽出し，renmae"""
    df_new = df.copy()
    df_new = df_new[cols_rename.keys()]
    df_new = df_new.rename(columns=cols_rename)
    return df_new

def get_items_by_genre(graph: List, genre: str) -> List:
    """graphから所定のgenreのアイテム群を取得"""
    items = [x for x in graph if "genre" in x.keys() and x["genre"] == genre]
    return items

def get_id_from_uri(uri: Optional[str]) -> Optional[str]:
    """uri表記から末尾のidを取得"""
    if uri is np.nan:
        return None
    else:
        return uri.split("/")[-1]

def format_name(name: Optional[str]) -> Optional[str]:
    """nameから名称情報を抽出
    ccname, crtname, ocrtnameに利用"""
    if name is np.nan:
        return None
    if type(name) is dict:
        return None

    if type(name) is str:
        return name
    if type(name) is list:
        for x in name:
            if type(x) is str:
                return x

    raise Exception(f"No name in {name}!")

def format_price(price: Optional[Any]) -> Optional[int]:
    """priceを整形"""
    if price is np.nan:
        return None
    elif price == "JUMPガラガラウなかも":
        # M544740, 週刊少年ジャンプ 1971年 表示号数47
        # price = 'JUMPガラガラウなかも'
        return None
    elif price == "238p":
        # M542801, 週刊少年ジャンプ 2010年 表示号数42
        # price = '238p'
        return 238
    else:
        price_new = price.replace("円", "").replace("+税", "")
        return int(price_new)

def preprocess_df_mi(path: str, mcids: List) -> pd.DataFrame:
    """pathとmcidsからdf_miを構築"""
    filters = {
        "genre": ["雑誌巻号"],
        "isPartOf": [f"https://mediaarts-db.bunka.go.jp/id/{mcid}" for mcid in mcids],
    }
    mi = read_json_w_filters(path, "@graph.item", filters)
    df_mi = pd.DataFrame(mi)

    # 列を整理
    df_mi = format_cols(df_mi, COLS_MI)
    # mcidを取得
    df_mi["mcid"] = df_mi["mcid"].apply(lambda x: get_id_from_uri(x))
    # mcnameを設定
    df_mi["mcname"] = df_mi["mcid"].apply(lambda x: mcid2mcname.get(x, None))
    # dateでソート
    df_mi["date"] = pd.to_datetime(df_mi["date"])
    df_mi = df_mi.sort_values("date", ignore_index=True)
    # priceを整形
    df_mi["price"] = df_mi["price"].apply(lambda x: format_price(x))
    return df_mi

def preprocess_df_ce(path: str, miids: List) -> pd.DataFrame:
    """pathとmiidsからdf_ceを構築"""
    filters = {
        "genre": ["マンガ作品"],
        "isPartOf": [f"https://mediaarts-db.bunka.go.jp/id/{miid}" for miid in miids],
    }
    ce = read_json_w_filters(path, "@graph.item", filters)
    df_ce = pd.DataFrame(ce)

    # 列を整形
    df_ce = format_cols(df_ce, COLS_CE)
    # uri表記から各idを取得
    df_ce["ccid"] = df_ce["ccid"].apply(lambda x: get_id_from_uri(x))
    df_ce["miid"] = df_ce["miid"].apply(lambda x: get_id_from_uri(x))
    return df_ce

def get_crtname_from_text(text: Optional[str]) -> Optional[str]:
    """textからcrtnameを抽出"""
    # [原作]，［原作］等を削除
    text = re.sub("\［.*\］", "", text)
    text = re.sub("\[.*\]", "", text)
    # 全角/半角カギカッコが混ざったコーナーケース
    text = text.replace("［原作]", "")
    text = re.sub("\s*", "", text)
    return text

def get_crtnames_from_text(text: Optional[str]) -> Optional[List]:
    """textからcrtnameのリストを抽出する"""
    text = format_name(text)
    if text is np.nan or text is None:
        return None
    else:
        crtnames = re.split("[/&＆／]", text)
        crtnames = [get_crtname_from_text(c) for c in crtnames if len(c)]
        return crtnames

def preprocess_df_cc(path: str, ccids: List) -> pd.DataFrame:
    """df_ccの前処理"""
    cm106 = read_json(path)
    # 雑誌掲載ジャンルのアイテムを抽出
    cc = get_items_by_genre(cm106["@graph"], "雑誌掲載")
    # DataFrame化
    df_cc = pd.DataFrame(cc)
    # カラムを整理
    df_cc = format_cols(df_cc, COLS_CC)
    # ccidでフィルタ
    df_cc = df_cc[df_cc["ccid"].isin(ccids)].reset_index(drop=True)
    # ccnameを整形
    df_cc["ccname"] = df_cc["ccname"].apply(lambda x: format_name(x))
    # 著者名を取得
    df_cc["crtname"] = df_cc["crtname"].apply(lambda x: get_crtnames_from_text(x))
    # 原作者名を取得
    df_cc["ocrtname"] = df_cc["ocrtname"].apply(lambda x: get_crtnames_from_text(x))

    return df_cc

def read_csvs(pathes: List) -> pd.DataFrame:
    """pathesのcsvを順番に呼び出し，concat"""
    df_all = pd.DataFrame()
    for p in pathes:
        df = pd.read_csv(p)
        df_all = pd.concat([df_all, df], ignore_index=True)
    return df_all

def sort_by_date(df: pd.DataFrame, col_date: str) -> pd.DataFrame:
    """dfをcol_dateでソート"""
    df_new = df.copy()
    df_new[col_date] = pd.to_datetime(df_new[col_date])
    df_new = df_new.sort_values(col_date, ignore_index=True)
    return df_new

def cast_str_to_list(text: Optional[str]) -> Optional[str]:
    """str形式をリストに再変換
    絶対他に良い方法がある気がする"""
    if text is np.nan:
        return []
    text = text.replace("[", "").replace("]", "").replace("'", "").replace(" ", "")
    return text.split(",")

# miidとpage_startでソート

# 適切なpage_startとpage_endでフィルタ

filter_ps_pe = df_ce["page_start"] <= df_ce["page_end"]

filter_pe = df_ce["page_end"] <= MAX_PAGES

df_ce = df_ce[(filter_ps_pe & filter_pe)].reset_index(drop=True)
